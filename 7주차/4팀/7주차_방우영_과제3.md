import pandas as fubao
import matplotlib.pyplot as plt

# 판다들의 데이터
data = {
    '판다 이름': ['진진', '자이언트', '메이메이', '타이타이', '쑤닝'],
    '나이(년)': [5, 10, 8, 7, 9],
    '성별': ['암컷', '수컷', '암컷', '수컷', '수컷'],
    '건강 상태': ['건강', '건강', '경미한 질병', '건강', '경미한 질병'],
    '대나무(%)': [60, 50, 40, 55, 70],
    '과일(%)': [20, 30, 40, 20, 15],
    '워토우(%)': [5, 10, 10, 15, 10],
    '콩(%)': [5, 5, 5, 5, 5],
    '장미(%)': [5, 5, 5, 5, 5],
    '죽순(%)': [5, 5, 5, 5, 5]
}

df = fubao.DataFrame(data)

# 판다 이름과 각 음식의 선호도 추출
food_columns = ['대나무(%)', '과일(%)', '워토우(%)', '콩(%)', '장미(%)', '죽순(%)']
pandas = df['판다 이름']

# 시각화
fig, ax = plt.subplots(figsize=(10, 6))
index = 0
for panda in pandas:
    ax.barh(df.columns[4:], df.loc[index, food_columns], label=panda)
    index += 1 

ax.set_xlabel('선호 비율 (%)')
ax.set_title('판다들의 음식 선호도')
ax.legend(title='판다 이름')
plt.show()
